    M140 S0 # Turn off the bed
    M141  # Turn off the chamber
    G1 E-3 F1800 # Retract a little filament
    G91 # Change to relative positioning
    G0 Z5 F300 # Move the nozzle up 5 (move bed down 5mm)
    G90 # Return to absolute positioning
    WIPE # Use the nozzle wipe
    # Just let WIPE park the toolhead
